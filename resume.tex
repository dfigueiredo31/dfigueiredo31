%----------------------------------------------------------------------------------------
%	DOC CLASS AND PAGE STYLE
%----------------------------------------------------------------------------------------

\documentclass[a4paper,11pt]{article}
\pagestyle{empty} % Suppress headers and footers on all pages

%----------------------------------------------------------------------------------------
%	REQUIRED PACKAGES
%----------------------------------------------------------------------------------------

\usepackage{geometry} % Required for adjusting page dimensions and margins
\usepackage[parfill]{parskip}   % Remove paragraph indentation
\usepackage{array}              % Required for bold tabular columns
\usepackage{ifthen}             % Required for ifthenelse statements
\usepackage{titlesec}           % Title formatting
\usepackage{xstring}

%----------------------------------------------------------------------------------------
%	MARGINS
%----------------------------------------------------------------------------------------

\geometry{
	top=1cm, % Top margin
	bottom=1cm, % Bottom margin
	left=1cm, % Left margin
	right=1cm, % Right margin
	%showframe, % Uncomment to show how the type block is set on the page
}

\setlength{\parskip}{0.2em}

%----------------------------------------------------------------------------------------
%	SECTION TITLES
%----------------------------------------------------------------------------------------

\setcounter{secnumdepth}{0}   %sets section count to 0

%section and subsection formatting
\titleformat{\section}        
  {\normalfont\huge\bfseries\center\uppercase}
  {}
  {}
  {}

\titleformat{\subsection}
  {\normalfont\large\bfseries}
  {\subsectiontitle \thesubsection}
  {}
  {}
  [{\titlerule[1pt]}]

%section and subsection tittle spacing
\titlespacing{\section}
  {0em}
  {0em}
  {0em}

\titlespacing{\subsubsection}
  {0em}
  {0.5em}
  {-0.2em}

%----------------------------------------------------------------------------------------
%	LISTS
%----------------------------------------------------------------------------------------

\renewenvironment{itemize}{
  \begin{list}
    {-}
    {
      \setlength{\leftmargin}{1em}
      \setlength{\itemsep}{-0.3em}
      \setlength{\parskip}{0em}
      \setlength{\parsep}{0em}
    }
}{
  \end{list}
}

%----------------------------------------------------------------------------------------
%	DOCUMENT BODY
%----------------------------------------------------------------------------------------

\begin{document}

  \begin{center}
    \MakeUppercase{\huge \textbf{$name$}}\\$role$
  \end{center}

    $body$

\end{document}
